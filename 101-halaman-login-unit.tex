\section*{1. Halaman Login}

\addcontentsline{toc}{section}{1. Halaman Login}

\noindent Halaman Login merupakan halaman awal dari aplikasi SOP Kabupaten Cianjur. Berikut tahapan 
memasuki halaman landing page: 

\begin{enumerate}
    \item Pastikan PC/Komputer terhubung dengan koneksi internet. 
    \item Buka browser seperti chrome, edge, ataupun firefox. 
    \item Buka link http://128.199.171.208/probis-cianjurkab/login
    \item Berikut tampilan halaman Login.
    
    
    \begin{figure}[H]
        \centering
        \includegraphics[width=13.45cm]{media/login/1-halaman-login.png}
        \caption{Tampilan Halaman Login}
        \label{fig:bagan-alir}
    \end{figure}

    \item Mengisi Username dan Password
    
    \item Klik tombol Login
    
    \item Jika login sebagai Role Perangkat Daerah maka akan diarahkan ke halaman Dashboard Unit Kerja dengan tampilan berikut:
    \begin{figure}[H]
        \centering
        \includegraphics[width=12.45cm]{media/dashborad-unit-kerja/1-halaman-dashboard-probis-unit.png}
        \caption{Tampilan Halaman Dashboard}
        \label{fig:bagan-alir}
    \end{figure}

    \item Jika salah memasukkan Username atau Password maka akan muncul pesan seperti berikut:
    \begin{figure}[H]
        \centering
        \includegraphics[width=12.45cm]{media/login/2-pesan-gagal.png}
        \caption{Tampilan Pesan Gagal Login}
        \label{fig:bagan-alir}
    \end{figure}
    
\end{enumerate}
